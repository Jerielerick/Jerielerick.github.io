const express = require('express');
const session = require('express-session');
const cookieParser = require('cookie-parser');

class Server {
    constructor() {
        this.app = express();
        this.port = process.env.PORT;

        this.middlewares();
        this.routes();
    }

    middlewares() {
        this.app.use(express.static('public'));
        this.app.set('view engine', 'ejs');

        this.app.use(cookieParser());

        this.app.use(session({
            secret: "amar",
            saveUninitialized: true,
            resave: true
        }));
    }

    routes() {
        this.app.get('/hola', (req, res) => {
            if (res.session.user) {
                res.send("iniciasre secion " + req.session.user.nam);
            } else {
                res.send("NO has inisado secion");
            }
        });

        this.app.get('/login', (req, res) => {
            let nombre = req.query.nombre;
            let password = req.query.password;

            //conexion bd
            let mysql = require('mysql');

            let con = mysql.createConnection({
                host: "localhost",
                user: "root",
                password: "",
                database: "appweb"
            });

            con.connect((err) => {
                if (err) throw err;
                console.log("Connected!");

                let sql = "SELECT * FROM usuarios WHERE nombre_usuario ='" + nombre + "'";

                // SQL y usar parámetros para evitar SQL injection

                console.log("Consulta enviada: " + sql);
                con.query(sql, (err, result) => {
                    if (err) throw err;

                    if (result.length > 0) {
                        // Usuario encontrado, comparar contraseñas
                        if (password === result[0].password) {
                            console.log("Inicio de sesión exitoso");
                            res.render('quinielas');
                        } else {
                            console.log("Contraseña incorrecta");
                            res.send('Contraseña incorrecta');
                        }
                    } else {
                        console.log("Usuario no encontrado");
                        res.send('Usuario no encontrado');
                    }
                });
            });
        });

                host: "localhost",
                user: "root",
                password: "",
                database: "appweb"
            });

            con.connect((err) => {
                if (err) throw err;
                console.log("Connected!");

                let sql = "INSERT INTO ialumnos (`firstname`, `lastname`, `edad`, `country`, `password`, `email`, `phone`) VALUES ('" + firstname + "','" + lastname + "', " + age + " , '" + country + "','" + password + "','" + email + "','" + phone + "')";

                con.query(sql, (err, result) => {
                    if (err) throw err;

                    console.log("1 record inserted");

                    // Obtener el idStudent del resultado de la inserción
                    let insertedId = result.insertId;

                    res.render('registrado', {

                    let insertedId = result.insertId;

            let mysql = require('mysql');

            let con = mysql.createConnection({
                host: "localhost",
                user: "root",
                password: "",
                database: "appweb"
            });

            con.connect((err) => {
                if (err) throw err;
